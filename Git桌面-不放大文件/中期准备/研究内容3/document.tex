\documentclass[border=2pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{3d,calc}

\begin{document}
	\begin{tikzpicture}[line join=round, line cap=round, >=latex]
		% 为了演示，这里手动画一个简易的“抛物面网格”
		% 1. 画底部横线和靠近观察者的边线
		\draw[thick] (0,0) -- (4,0);
		\draw[thick] (0,0) -- (2,2.5);
		
		% 2. 画右侧竖线
		\draw[thick] (4,0) -- (4,2);
		
		% 3. 画左侧竖线
		\draw[thick] (2,2.5) -- (2,4);
		
		% 4. 画顶点虚线
		%    这里可按照实际尺寸、曲面需求来调整弧度
		\draw[dashed] plot [domain=0:2] (\x, {2*\x-0.5*\x*\x+2.5});
		
		% 5. 画抛物面内部的网格虚线
		% 简单示例：从左到右画若干弧线
		\foreach \x in {0.5,1,1.5}{
			\draw[dashed]
			plot [domain=0:\x] (\t, {2*\t-0.5*\t*\t+2.5})
			-- ++( {4-\x}, { -(\x/2.0)*1.2 } );
		}
		% 也可以再添加从后往前的网格线
	\end{tikzpicture}
\end{document}
